Mutex互斥锁得实现


package main

import (
   "fmt"
   "sync"
)

var num int
var mtx sync.Mutex
var wg sync.WaitGroup

func add() {
   mtx.Lock()
   defer mtx.Unlock()
   defer wg.Done()
   num += 1
}

func main() {

   for i := 0; i < 100; i++ {
      wg.Add(1)
      go add()
   }

   wg.Wait()

   fmt.Println("num:", num)

}
